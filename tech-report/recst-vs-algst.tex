This section describes how recursive session types \`a la
\citet{DBLP:journals/jfp/GayV10} can be embedded in algebraic protocols. To
differentiate the two different type languages we use ``algebraic types'' to
refer to session and protocol types as defined by
\citet{DBLP:journals/pacmpl/MordidoS0V23}, ``recursive types'' refer to
classical, end-recursive session types. Both notions also include the
corresponding functional types.


\subsection{A graph representation for recursive types}

Recursive session types are represented as a deterministic labelled transition
system (LTS) $\stlts$ where $V$ is a set of nodes or states, $L$ is the set of
transition labels containing at least $\Lone,\Ltwo\in L$, and ${\sttrans{}}
\subseteq V\times L\times V$ the transition relation. We may write
$v\sttrans\ell v'$ for $(v,\ell,v')\in{\sttrans{}}$, and $v\notrans$ if there
are no $\ell,v'$ such that $v\sttrans\ell v'$. There is a designated initial
state $v_0\in V$. The labelling function $\lab:V\to \LSet$ assigns a label to
every state. The subset $\LS = \{\LBranch,\LChoice,\LSend,\LRecv,\LEnd\}$ is
called the set of session type labels. A node $v\in V$ with $\lab(v)\in\LS$ is
called a session type node. The transition relation must be consistent with the
transition relation. Every node $v\in V$ must satisfy the conditions below.
%
\begin{enumerate}
  %
  \item \label{tylts:arity0} If $\lab(v) \in \{\LEnd,\LUnit\}$ it must hold
    that $v\notrans$.
  %
  \item \label{tylts:arity2} If $\lab(v) \in \{\LSend,\LRecv,\LFunc\}$ there
    must be $v_1,v_2\in V$ with $v \sttrans\Lone v_1$ and $v \sttrans\Ltwo
    v_2$, and for all $v \sttrans\ell v'$ it must hold that
    $(\ell,v')\in\{(\Lone,v_1),(\Ltwo,v_2)\}$.
  %
  \item If $\lab(v)\in\{\LChoice,\LBranch\}$ there must be at least one
    transition $v\sttrans\ell v'$, and for all transitions $v \sttrans{\ell'}
    v''$ it must hold that $\ell'\not\in\{1,2\}$.
  %
  \item If $\lab(v)\in\LS$, for all $v\sttrans\ell v'$ with $\ell \ne \Lone$ it
    must hold that $\lab(v')\in\LS$.
  %
  \item \label{tylts:cycles} If there is a cycle $v \sttrans{\ell} v_1
    \sttrans{\ell_1} \cdots v_n \sttrans{\ell_n} v_1$ it must hold that
    $\lab(v)\in\LS$ and $\ell\ne\Lone$.
  %
  \item If $v\sttrans\ell v'$ and $v\sttrans\ell v''$ it must hold that
    $v'=v''$.
  %
\end{enumerate}
%
Requirement (\ref{tylts:cycles}) restricts the kind of cycles that may appear.
The only cycles that are allowed must consists solely of session type nodes and
the transition labels must not be $\Lone$. The restriction serves two goals:
only sessions types are allowed to be recursive and we avoid the dualization
paradox where recursive session types use recursion variables in the payload of
messages~\citep{BernardiH16,DBLP:journals/corr/abs-2004-01322}.

Given an LTS $(V,L,\sttrans{},v_0)$ the function $\stty{{\,\cdot\,}}$ produces
a type with the grammar below. In particular, if $\lab(v)\in\LS$ for some $v\in
V$, the result of $\stty{v}$ is a type according to the grammar $S$.
%
\begin{align*}
  %
  T &\grmeq \dots \\
  %
  S &\grmeq \dots
  %
\end{align*}
%
% $\eta$ is a partial function from nodes to recursion variables. We write
% $\eta(v)\downarrow$ if $\eta$ is defined at $v$, and $\eta(v)\uparrow$ if not.
% $\emptyset$ denotes the function that is not defined at any point and
% $\eta[v\mapsto\alpha]$ the partial function defined as $\alpha$ at point $v$
% and defined according to $\eta$ at any other point.

\[
  \stty{v} = \begin{cases*}
    v & if
      $v \in \eta$
    \\
    \mu v. \SBranchChoice\star*{{\ell: \stty[\eta\cup\{v\}]{v'}}}_{\ell\in\ConSet'} & if
      %$\eta(v)\uparrow$,
      $v \not\in \eta$,
      $\star = \lab(v) \in \{\LBranch,\LChoice\}$,
      %$\alpha$ fresh,
      %for all $\ell,v'$ with $v \sttrans\ell v'$
      $\ConSet' = \{ \ell\in\ConSet \mid \exists v'\!.\, v \sttrans\ell v' \}$
    \\
    \mu v. \SSendRecv\sharp{\stty[\emptyset]{v_1}} \stty[\eta\cup\{v\}]{v_2} & if
      %$\eta(v)\uparrow$,
      $v \not\in \eta$,
      $\sharp = \lab(v) \in \{\LSend,\LRecv\}$,
      %$\alpha$ fresh,
      $v \sttrans{\Lone} v_1, v \sttrans{\Ltwo} v_2$
    \\
    \stty[\emptyset]{v_1} \to \stty[\emptyset]{v_2} & if
      $\lab(v) = \LFunc$,
      $v \sttrans{\Lone} v_1, v \sttrans{\Ltwo} v_2$
    \\
    \TEnd & if
      $\lab(v) = \LEnd$
  \end{cases*}
\]
%

\subsection{Subtyping}

We define a subtyping relation on the LTS representation and prove it sound
with respect to the classical subtyping relation for session types
\citep{DBLP:journals/acta/GayH05,DBLP:journals/jfp/GayV10}.

Subtyping is defined on the transition relation of the LTS. Given two recursive
processes $T = (V, L, \sttrans{}_T, v_0, \lab)$ and $U = (V, L,
\sttrans{}_U, v_0, \lab)$ over the same set of nodes and labels, and with the
same labelling function, we define $T \SubLTS U$ by
$\SubNode{v_0}(\sttrans{}_{T}, \sttrans{}_{U})$, where $v_0$ is the
initial node. The binary relation $\SubNode{v}(\sttrans{~}_{A},
\sttrans{~}_{B})$ with parameter $v \in V$ is defined via a case split on the
label of $v$.
%
\begin{itemize}

  \item \textbf{Case $\lab(v) = \LBranch$:} For every transition $v \sttrans{\ell}_{A} v'$,
        there must be a matching transition $v \sttrans{\ell}_{B} v'$ such that
        $\SubNode{v'}(\sttrans{}_{A}, \sttrans{}_{B})$.

  \item \textbf{Case $\lab(v) = \LChoice$:} For every transition $v \sttrans{\ell}_{B} v'$,
        there must be a matching transition $v \sttrans{\ell}_{A} v'$ such that
        $\SubNode{v'}(\sttrans{}_{A}, \sttrans{}_{B})$.

  \item \textbf{Case $\lab(v) = \LSend$:} There must be matching transitions
        $v \sttrans{\Lone}_{A} v'_1$, $v \sttrans{\Lone}_{B} v'_1$ and
        $v \sttrans{\Ltwo}_{A} v'_2$, $v \sttrans{\Ltwo}_{B} v'_2$ such that
        $\SubNode{v'_1}(\sttrans{}_{B}, \sttrans{}_{A})$ and
        $\SubNode{v'_2}(\sttrans{}_{A}, \sttrans{}_{B})$.

  \item \textbf{Case $\lab(v) = \LRecv$:} There must be matching transitions
        $v \sttrans{\Lone}_{A} v'_1$, $v \sttrans{\Lone}_{B} v'_1$ and
        $v \sttrans{\Ltwo}_{A} v'_2$, $v \sttrans{\Ltwo}_{B} v'_2$ such that
        $\SubNode{v'_1}(\sttrans{}_{A}, \sttrans{}_{B})$ and
        $\SubNode{v'_2}(\sttrans{}_{A}, \sttrans{}_{B})$.

  \item \textbf{Case $\lab(v) = \LFunc$:} There must be matching transitions
        $v \sttrans{\Lone}_{A} v'_1$, $v \sttrans{\Lone}_{B} {v'_1}$ and
        $v \sttrans{\Ltwo}_{A} v'_2$, $v \sttrans{\Ltwo}_{B} {v'_2}$ such that
        $\SubNode{v'_1}(\sttrans{}_{B}, \sttrans{}_{A})$ and
        $\SubNode{v'_2}(\sttrans{}_{A}, \sttrans{}_{B})$.

  \item \textbf{Case $\lab(v) = \LEnd$:} The relation holds trivially.
\end{itemize}

For $\LChoice$, the direction of the implication requires the subtype to
provide at least all the transitions offered by the supertype, meaning the
smaller type can support more labels. Conversely, for $\LBranch$, the subtype
is restricted to taking only transitions permitted by the supertype, meaning
the smaller type must have fewer or an equal amount of labels. Furthermore, for
$\LSend$ and $\LRecv$, the continuation of the first argument (label $\Lone$)
is contravariant and covariant, respectively. Similarly, the argument (label
$\Lone$) for $\LFunc$ is contravariant. Finally, $\LEnd$ holds trivially
because it admits no transitions.

\subsection{Recursive types as protocols}

This section describes how a recursive session type is translated to a valid
protocol type. The types are assumed to be available in their LTS
representation described previously. A node in the LTS labelled by an element
of $\LS$ is translated to a protocol type. Other nodes are translated to types
of kind $\kindm$ by embedding protocol type $\typec P$ as session types
$\typec{!P.\TEnd}$.

\begin{lstlisting}[
  float,
  caption={Supporting protocols used in the translation to algebraic protocols.},
  label=lst:translation-protocols,
  emph={Eps,Seq,Rep},
  frame=single
]
protocol Eps      = Eps
protocol Seq p q  = Seq p q
protocol Rep p    = Rep p (Rep p)
\end{lstlisting}

The translation makes use of the three singleton protocols shown in
\cref{lst:translation-protocols}. Protocol $\PEps$ is operationally a no-op,
$\TPApp{\TPApp{\PSeq}{p}}{q}$ sequences two protocols $\typec p$ and $\typec
q$, and $\TPApp\PRep p$ repeats protocol $\typec p$ ad infinitum. They are used
to encode recursive session types that do not contain branch or choice
constructors. For example, the session type the session type
$S_{\rightleftharpoons} = \mu\alpha.\SSend\TInt\SRecv\TInt\alpha$ will be
encoded as
$\typec{!\TPApp\PRep{(\TPApp{\TPApp\PSeq{\ppos\TInt}}{(\TPApp{\TPApp\PSeq{\pneg\TInt}}{\PEps})})}.\TEnd}$.

All other session types are encoded as a potentially empty sequence of sends
and receives, followed optionally by a single send or receive of a global
$\PDispatch$ protocol synthesized by the translation, and terminated by
$\TEnd$. In this case, any recursion is ``hidden'' inside the $\PDispatch$
protocol. The translation of whole programs only synthesizes one such protocol.
Protocol subtyping is used to restrict the set of constructors that are
actually valid. For example the session type $\SSend\TInt\mu\alpha. \SChoice{
\textsf{Add}:{\SSend\TInt\alpha}, \textsf{Result}:{\SRecv\TInt\TEnd} }$ is
encoded as $\typec{!\TInt.!\PDispatch[\textsf{Add},\textsf{Result}].\TEnd}$
where $\PDispatch$ has at least the following constructors:
%
\begin{lstlisting}[emph={Dispatch,Int,Eps},emph={[2]Add,Result},emphstyle={[2]\color{ACMDarkBlue}}]
protocol Dispatch = ...
                  | Add Int Dispatch[Add, Result]
                  | Result -Int Eps
\end{lstlisting}

The function $\textsc{seq}^\sharp : (\{!,?\} \times \typeset\kindt)^* \to
\typeset\kindp$ where $\sharp \in \{!,?\}$ turns a sequence of types to send or
receive into a protocol type using the $\PSeq$ and $\PEps$ protocols.
%
\[
  \textsc{seq}^\sharp(x) = \begin{cases*}
    \PEps & if $x$ is the empty sequence \\
    \TPApp{\TPApp \PSeq {\phantom{\pneg{}}T}}{\textsc{seq}^\sharp(x')}
      & if $x=(\sharp,\typec T)x'$ \\
    \TPApp{\TPApp \PSeq {\pneg T}}{\textsc{seq}^\sharp(x')}
      & if $x=(\sharp',\typec T)x'$ where $\sharp\ne\sharp'$
  \end{cases*}
\]

\[
  \algty(v) = \begin{cases*}
    \TFun[] {\algty(v_1)} {\algty(v_2)} & if
      $\lab(v) = \LFunc$,
      $v \sttrans{\Lone} v_1, v \sttrans{\Ltwo} v_2$,
    \\
    \typec{\sharp \TPApp\PRep T.\TEnd} & if
      $v_1 \sttrans\Ltwo \cdots v_n \sttrans\Ltwo v_1$ with $v=v_1, \sharp=\lab(v)$
    \\& \quad and for all $1 \leq i \leq n$: $v_i\sttrans\Lone v'_i$
    \\& \quad such that $\typec T = \cramped{\textsc{seq}^\sharp}((\lab(v_1), \algty(v'_1))\dots(\lab(v_n), \algty(v'_n)))$
    \\
    \typec{\sharp \algty(v_1) . \algty(v_2)} & if
      $\sharp = \lab(v) \in \{\LSend,\LRecv\}$,
      $v \sttrans{\Lone} v_1, v \sttrans{\Ltwo} v_2$
    \\
    \typec{\sharp \PDispatch[I].\TEnd} & if
      $\lab(v) \in \{\LChoice,\LBranch\}$ with
        $\sharp=\typec ! \Leftrightarrow \lab(v) = \LChoice$,
        $\sharp=\typec ? \Leftrightarrow \lab(v) = \LBranch$
    \\& \quad where $I = \{\ell\in L \mid \exists v'. v\sttrans\ell v'\}$
  \end{cases*}
\]
