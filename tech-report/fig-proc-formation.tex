    {\isProc[\Gamma_1,\Gamma_2]{\parp{p_1}{p_2}} }

    \infrule{P-New}
    {\typeAgainst[\Empty]{T}{\kinds} \\ \isProc[\Gamma, \ebind x{\Normal[+]{T}} , \ebind y{\Normal[-]{T}}]{p}}
    {\isProc{\newp{x}{y}{p}} }
